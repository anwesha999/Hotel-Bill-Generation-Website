    <script type="text/javascript">
function Sum(){
    var arr = document.getElementsByName('num');
    var to=0;
    for(var i=0;i<arr.length;i++){
        if(parseInt(arr[i].value))
            to += parseInt(arr[i].value);
    }
    document.getElementById('totalsum').value = to;
}
